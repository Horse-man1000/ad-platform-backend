🔐 5. Add Token Encryption Utility
Create src/utils/crypto.js:

import crypto from "crypto";

const ALGO = "aes-256-gcm";

export function encrypt(text) {
  const iv = crypto.randomBytes(16);
  const key = Buffer.from(process.env.ENCRYPTION_KEY);
  const cipher = crypto.createCipheriv(ALGO, key, iv);

  let encrypted = cipher.update(text, "utf8", "hex");
  encrypted += cipher.final("hex");
  const tag = cipher.getAuthTag().toString("hex");

  return `${iv.toString("hex")}:${tag}:${encrypted}`;
}

export function decrypt(enc) {
  const [ivHex, tagHex, data] = enc.split(":");
  const iv = Buffer.from(ivHex, "hex");
  const key = Buffer.from(process.env.ENCRYPTION_KEY);
  const tag = Buffer.from(tagHex, "hex");

  const decipher = crypto.createDecipheriv(ALGO, key, iv);
  decipher.setAuthTag(tag);

  let str = decipher.update(data, "hex", "utf8");
  str += decipher.final("utf8");
  return str;
}

🧠 7. Create API Endpoints
Here are the minimal working API files.
src/api/clients.js
import express from "express";
import { prisma } from "../utils/prisma.js";

const router = express.Router();

router.post("/", async (req, res) => {
  const client = await prisma.client.create({
    data: req.body
  });
  res.json(client);
});

router.get("/", async (req, res) => {
  const clients = await prisma.client.findMany();
  res.json(clients);
});

export default router;
``
src/api/tokens.js
import express from "express";
import { prisma } from "../utils/prisma.js";
import { encrypt } from "../utils/crypto.js";

const router = express.Router();

router.post("/", async (req, res) => {
  const { clientId, platform, accessToken, refreshToken, expiresAt } = req.body;

  const token = await prisma.clientToken.upsert({
    where: { clientId_platform: { clientId, platform } },
    update: {
      accessToken: encrypt(accessToken),
      refreshToken: refreshToken ? encrypt(refreshToken) : null,
      expiresAt: expiresAt ? new Date(expiresAt) : null
    },
    create: {
      clientId,
      platform,
      accessToken: encrypt(accessToken),
      refreshToken: refreshToken ? encrypt(refreshToken) : null,
      expiresAt: expiresAt ? new Date(expiresAt) : null
    }
  });

  res.json(token);
});

export default router;
``
🚦 8. Main App Entry (Express Server)
Create src/app.js:
import express from "express";
import cors from "cors";
import clients from "./api/clients.js";
import tokens from "./api/tokens.js";

const app = express();
app.use(cors());
app.use(express.json());

app.use("/clients", clients);
app.use("/tokens", tokens);

app.get("/", (req, res) => {
  res.send("Ad Platform API is running");
});

app.listen(3000, () => console.log("Server running on http://localhost:3000"));
🛠️ 9. Start the Server
Run:
node src/app.js
